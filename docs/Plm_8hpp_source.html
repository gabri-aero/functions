86 Plm(int l_max, double theta, bool derivatives = false, bool second_derivatives = false) : l_max(l_max), _Nlm(Nlm(l_max)), theta(theta)
87 {
88 // Allocate ALFs
89 this->_Plm = new double[((l_max + 1) * (l_max + 2)) / 2];
90 // Define constants for FOID recursion
91 double *a = new double[((l_max + 1) * (l_max + 2)) / 2];
92 double *b = new double[((l_max + 1) * (l_max + 2)) / 2];
93 for (int l = 0; l <= l_max; l++)
94 {
95 for (int m = 0; m < l; m++)
96 {
97 a[lm_idx(l, m)] = sqrt((2 * l - 1.0) * (2 * l + 1) / ((l - m) * (l + m)));
98 b[lm_idx(l, m)] = l - m != 1 ? sqrt(((2 * l + 1.0) * (l + m - 1) * (l - m - 1)) /
99 ((l - m) * (l + m) * (2 * l - 3)))
100 : 0;
101 }
102 }
103 // Define cosine, sine
104 double t = cos(theta);
105 double u = sin(theta);
106 // Define P00
107 _Plm[0] = 1;
108 // Define P11
109 if (l_max > 0)
110 _Plm[lm_idx(1, 1)] = sqrt(3) * u;
111 // Recursion for sectorial polynomials
112 for (int l = 2; l <= l_max; l++)
113 {
114 _Plm[lm_idx(l, l)] = sqrt((2 * l + 1.0) / (2 * l)) * u * _Plm[lm_idx(l - 1, l - 1)];
115 }
116 // Recursion for terms below diagonal
117 for (int m = 0; m < l_max; m++)
118 { // Fix order
119 // Now increase degree
120 int l = m + 1;
121 // Terms right below the diagonal
122 _Plm[lm_idx(l, m)] = a[lm_idx(l, m)] * t * _Plm[lm_idx(l - 1, m)];
123 // Other terms
124 for (int l = m + 2; l <= l_max; l++)
125 {
126 _Plm[lm_idx(l, m)] = a[lm_idx(l, m)] * t * _Plm[lm_idx(l - 1, m)] -
127 b[lm_idx(l, m)] * _Plm[lm_idx(l - 2, m)];
128 }
129 }
130 // Delete allocated data
131 delete[] a;
132 delete[] b;
133 // Compute derivatives
134 if (derivatives)
135 {
136 // Allocate derivatives
137 _dPlm = new double[((l_max + 1) * (l_max + 2)) / 2];
138 // Constant for derivatives recursion
139 double *f = new double[((l_max + 1) * (l_max + 2)) / 2];
140 for (int l = 0; l <= l_max; l++)
141 {
142 for (int m = 0; m <= l; m++)
143 {
144 f[lm_idx(l, m)] = sqrt((l * l - m * m) * (2 * l + 1) / (2 * l - 1.0));
145 }
146 }
147 // Sectorial terms
148 for (int m = 0; m <= l_max; m++)
149 {
150 _dPlm[lm_idx(m, m)] = m * t / u * _Plm[lm_idx(m, m)];
151 }
152 // Terms below diagonal
153 for (int l = 1; l <= l_max; l++)
154 {
155 for (int m = 0; m < l; m++)
156 {
157 _dPlm[lm_idx(l, m)] =
158 1.0 / u *
159 (l * t * _Plm[lm_idx(l, m)] - f[lm_idx(l, m)] * _Plm[lm_idx(l - 1, m)]);
160 }
161 }
162
163 // Compute 2nd order derivatives
164 if (second_derivatives)
165 {
166 // Allocate 2nd order derivatives
167 _ddPlm = new double[((l_max + 1) * (l_max + 2)) / 2];
168 // Sectorial terms
169 for (int m = 0; m <= l_max; m++)
170 {
171 _ddPlm[lm_idx(m, m)] =
172 (m - 1) * t / u * _dPlm[lm_idx(m, m)] - m * _Plm[lm_idx(m, m)];
173 }
174 // Terms below diagonal
175 for (int l = 1; l <= l_max; l++)
176 {
177 for (int m = 0; m < l; m++)
178 {
179 _ddPlm[lm_idx(l, m)] = 1.0 / u *
180 ((l - 1) * t * _dPlm[lm_idx(l, m)] -
181 f[lm_idx(l, m)] * _dPlm[lm_idx(l - 1, m)]) -
182 l * _Plm[lm_idx(l, m)];
183 }
184 }
185 }
186 else
187 {
188 _ddPlm = nullptr;
189 }
190 // Delete allocated data
191 delete[] f;
192 }
193 else
194 {
195 _dPlm = nullptr;
196 }
197 };
